106 void m_duplicate(const D4Attributes &src) {
107 D4AttributesCIter i = src.d_attrs.begin();
108 while (i != src.d_attrs.end()) {
109 d_attrs.push_back(new D4Attribute(**i++)); // deep copy
110 }
111 }

123 }
