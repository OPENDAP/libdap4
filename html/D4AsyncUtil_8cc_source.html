291string D4AsyncUtil::getRejectReasonCodeString(RejectReasonCode code) {
292
293 string codeStr;
294 switch (code) {
295 case TIME:
296 codeStr = "time";
297 break;
298
299 case UNAVAILABLE:
300 codeStr = "unavailable";
301 break;
302
303 case PRIVILEGES:
304 codeStr = "privileges";
305 break;
306
307 case OTHER:
308 codeStr = "other";
309 break;
310
311 default:
312 throw InternalErr(__FILE__, __LINE__,
313 "D4AsyncUtil::getRejectReasonCodeString() - Unrecognized reject_reason_code.");
314 }
315
316 return codeStr;
317}
