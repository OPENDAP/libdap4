81class mp_lock_guard {
82public:
83 enum class operation {
84 read, write
85 };
86private:
87 int d_fd = -1;
88 bool d_locked = false;
89 bool d_released = false; // Use this so instances can go out of scope without releasing the lock.
90 operation d_op;
91 const mp_lock_guard_logger &d_logger;
92
93 void m_get_lock() {
94 if (d_op == operation::write) {
95 struct flock lock{};
96 lock.l_type = F_WRLCK;
97 lock.l_whence = SEEK_SET;
98 lock.l_start = 0;
99 lock.l_len = 0;
100 lock.l_pid = getpid();
101 if (fcntl(d_fd, F_SETLKW, &lock) == -1) {
102 d_logger.error("Could not write lock the cache-control file: " + get_errno());
103 }
104 }
105 else {
106 struct flock lock{};
107 lock.l_type = F_RDLCK;
108 lock.l_whence = SEEK_SET;
109 lock.l_start = 0;
110 lock.l_len = 0;
111 lock.l_pid = getpid();
112 if (fcntl(d_fd, F_SETLKW, &lock) == -1) {
113 d_logger.error("Could not read lock the cache-control file: " + get_errno());
114 }
115 }
116 d_locked = true;
117 }
118
119 friend class mp_lock_guard_test;
120
121public:
122 mp_lock_guard() = delete;
123
124 mp_lock_guard(const mp_lock_guard &) = delete;
125
126 mp_lock_guard &operator=(const mp_lock_guard &) = delete;
127
134 mp_lock_guard(int fd, operation op, const mp_lock_guard_logger &logger = mp_lock_guard_logger_default())
135 : d_fd(fd), d_op(op), d_logger(logger) {
136 m_get_lock();
137 }
138
146 ~mp_lock_guard() {
147 if (!d_released && d_locked) {
148 struct flock lock{};
149 lock.l_type = F_UNLCK;
150 lock.l_whence = SEEK_SET;
151 lock.l_start = 0;
152 lock.l_len = 0;
153 lock.l_pid = getpid();
154 if (fcntl(d_fd, F_SETLK, &lock) == -1) {
155 d_logger.log("Could not unlock the cache-control file: " + get_errno());
156 }
157 }
158 }
159
164 void release() { d_released = true; }
165
169 static void unlock(int fd, const mp_lock_guard_logger &logger = mp_lock_guard_logger_default()) {
170 struct flock lock{};
171 lock.l_type = F_UNLCK;
172 lock.l_whence = SEEK_SET;
173 lock.l_start = 0;
174 lock.l_len = 0;
175 lock.l_pid = getpid();
176 if (fcntl(fd, F_SETLK, &lock) == -1) {
177 logger.error("Could not unlock the cache-control file: " + get_errno());
178 }
179 }
180};
